    def wait(timeout) 
      total = 0;
      interval = 0.2;
      
      if !block_given?
        sleep(timeout)
        return
      end
      
      while (total < timeout)
        sleep(interval);
        total += interval;
        begin 
          return if yield
        rescue Exception=>e 
          puts e
        end
      end          
    end
