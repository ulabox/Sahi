        def test_focus() 
                @browser.navigate_to(@base_url  + "/demo/focusTest.htm")
                @browser.textbox("t2").focus()
                assert_equal("focused", @browser.textbox("t1").value)
                @browser.textbox("t2").remove_focus()
                assert_equal("not focused", @browser.textbox("t1").value)
                @browser.textbox("t2").focus()
                assert_equal("focused", @browser.textbox("t1").value)         
        end
